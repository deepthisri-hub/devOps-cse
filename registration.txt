  <script>
    // Populate age dropdown from 1 to 30
    const ageSelect = document.getElementById("age");
    for (let i = 1; i <= 30; i++) {
      ageSelect.innerHTML += `<option value="${i}">${i}</option>`;
    }

    // Form submission handler
    document.getElementById("registrationForm").addEventListener("submit", function (e) {
      e.preventDefault();

      const requiredFields = ["name", "email", "phone", "college", "age", "dob", "event"];
      for (let id of requiredFields) {
        if (!document.getElementById(id).value.trim()) {
          alert("Please fill out all fields.");
          return;
        }
      }

      const gender = document.querySelector('input[name="gender"]:checked');
      const year = document.querySelector('input[name="year"]:checked');

      if (!gender || !year) {
        alert("Please select gender and year of study.");
        return;
      }

      document.getElementById("successMessage").style.display = "block";
      this.style.display = "none";
    });
  </script>
